# cppcheck-suppressions.txt
#
# Authoritative registry of every cppcheck suppression in the linkdb
# CI pipeline.  Spec-0.27.5 §3.4 (suppression maintenance rules)
# requires every suppression to have a one-line "Reason:" comment
# explaining why the cppcheck warning is wrong or unactionable.
#
# Implementation note: the suppressions are NOT loaded from this file
# by scripts/ci/run-cppcheck.sh -- they are passed as --suppress= CLI
# flags directly because cppcheck 2.13's --suppressions-list parser
# rejects mixed-comment files with "Failed to add suppression. No id."
# even when valid entries are present.  This file is kept as the
# human-facing registry; run-cppcheck.sh is the executable copy.
# Whenever you add a suppression, update both files in lockstep.
#
# Spec:    pgrac/specs/spec-0.27.5-static-analysis.md
# Design:  pgrac/docs/ci-static-analysis.md
# Baseline: pgrac/docs/ci-static-analysis-baseline-stage0.27.md
#
# Format: <id>[:<file>[:<line>]]   id is a cppcheck rule name.

# === Global PG framework noise (apply to all files) ===
# Reason: cppcheck flags every PG SRF function as "never used"
# because it cannot resolve the PG_FUNCTION_INFO_V1 macro registration.
unusedFunction
# Reason: cppcheck cannot resolve PG's hand-rolled include layout
# reliably with -I; missingInclude noise is not actionable.
missingInclude
missingIncludeSystem
# Reason: future inline /* cppcheck-suppress xxx */ removals should
# not fire unmatched-suppression warnings during the transition.
unmatchedSuppression

# === PG-upstream headers reachable via -I src/include ===
# Reason: cppcheck transitively follows -I src/include and warns on
# PG-upstream headers; spec-0.27.5 §1.2 limits scope to cluster code.
# PG header s_lock.h emits #error for unsupported-platform builds;
# cppcheck cannot evaluate the platform conditionals correctly.
preprocessorErrorDirective:src/include/storage/s_lock.h
# Reason: PG-upstream defensive null check; pg_list.h is not pgrac code.
nullPointerRedundantCheck:src/include/nodes/pg_list.h
# Reason: PG-upstream defensive null check; bufpage.h is not pgrac code.
nullPointerRedundantCheck:src/include/storage/bufpage.h
# Reason: PG-upstream xlogreader.h reachable via cluster_smgr.c -> smgr.h
# transitive include (added in stage 1.2); not pgrac code.
constParameterPointer:src/include/access/xlogreader.h
# Reason: spec-1.10.1 (F1 hardening) made cluster_startup_phase.h include
# storage/lwlock.h to expose ClusterPhaseSharedState; lwlock.h transitively
# pulls in port/atomics/generic.h.  Two PG-upstream findings surface there
# (constParameterPointer + integerOverflowCond on signed -INT_MIN); both
# are out of pgrac scope per spec-0.27.5 §1.2.
constParameterPointer:src/include/port/atomics/generic.h
integerOverflowCond:src/include/port/atomics/generic.h
# Reason: stage-1.3 cluster_shmem_register_region uses Assert(region != NULL);
# region->name pattern.  PG Assert is non-trapping to cppcheck, so it flags
# the later region != NULL check as redundant.  Same precedent as
# cluster_shared_fs_local.c suppression added in stage 1.1.
nullPointerRedundantCheck:src/backend/cluster/cluster_shmem.c

# Reason: spec-2.3 D1 cluster_ic_envelope.h uses pg_attribute_packed() macro
# from c.h to honor spec-2.0 §4 frozen offsets (uint64 epoch at offset 12 /
# scn at offset 20 are non-8-aligned naturally; without __attribute__((packed))
# the struct grows to 40 bytes).  cppcheck 2.13 cannot expand the PG macro
# and parses the trailing identifier `ClusterICEnvelope` as a syntax error.
# The compile-time ABI lock is the 4 StaticAssertDecl in cluster_ic_envelope.c
# (sizeof + 3 offsetof); cppcheck's syntax confusion is purely a macro-
# expansion gap, not a code defect.
syntaxError:src/include/cluster/cluster_ic_envelope.h

# Reason: spec-2.5 D1 cluster_cssd.h uses pg_attribute_packed() macro on
# ClusterCssdHeartbeatPayload (12-byte wire format).  Without packed the
# struct grows to 16 bytes (uint64 forces 8-byte struct alignment).
# cppcheck 2.20 cannot expand pg_attribute_packed() macro and reports
# syntaxError on the trailing identifier `ClusterCssdHeartbeatPayload`.
# Compile-time ABI lock = StaticAssertDecl(sizeof == 12) in cluster_cssd.c.
# Same pattern as cluster_ic_envelope.h precedent above (L67 in lessons SSOT).
syntaxError:src/include/cluster/cluster_cssd.h

# === cluster_unit standalone stubs ===
# Reason: standalone cluster unit tests define PG-core and pgrac stubs with
# signatures that intentionally mirror non-const production APIs.  cppcheck
# 2.20 flags many stub parameters as constParameterPointer candidates because
# the stub bodies are no-ops.  Changing those signatures would stop matching
# the production symbol contracts under test, so suppress this style-only
# finding for cluster_unit test stubs.
constParameterPointer:src/test/cluster_unit/test_cluster_*.c
